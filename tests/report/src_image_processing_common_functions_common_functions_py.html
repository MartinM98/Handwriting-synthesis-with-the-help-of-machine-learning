53def process_part(image: np.ndarray, x1: int, y1: int, x2: int, y2: int): 

54 """ 

55 Process a part of an image returning 

56 list containing locations of points/pixels 

57 which are black (have value 0) 

58 

59 Args: 

60 image (np.ndarray): the whole cv2 image. 

61 x1 (int): width value of the left edge of the selected part of the image. 

62 y1 (int): height value of the upper edge of the selected part of the image. 

63 x2 (int): width value of the right edge of the selected part of the image. 

64 y2 (int): height value of the lower edge of the selected part of the image. 

65 

66 Returns: 

67 points (list): list containing locations of points/pixels 

68 which are black (have value 0). 

69 """ 

70 points = [] 

71 for y in range(y1, y2 + 1): 

72 for x in range(x1, x2 + 1): 

73 point = [] 

74 if image[x, y] == 0: 

75 point.append(x) 

76 point.append(y) 

77 points.append(point) 

78 return points

96def get_dimensions(img: np.ndarray, n: int): 

97 """ 

98 Computes and returns steps that divides the image 

99 into n x n parts and returns dimensions 

100 of the image as well as the steps in both 

101 directions 

102 Args: 

103 img (np.ndarray): cv2 image. 

104 n (int): the number of subparts in each direction. 

105 

106 Returns: 

107 x_step (int): x step that is the horizontal 

108 length of the subpart of the image. 

109 width (int): The width of the image. 

110 y_step (int): y step that is the vertical length 

111 of the subpart of the image. 

112 height (int): The height of the image. 

113 """ 

114 width = img.shape[0] 

115 height = img.shape[1] 

116 x_step = math.ceil(width / n) 

117 y_step = math.ceil(height / n) 

118 return width, height, x_step, y_step

121def get_box(x: int, x_step: int, width: int, y: int, y_step: int, height: int): 

122 """ 

123 Computes and returns the next box based on initial x,y and steps. 

124 The functions truncates the points if they would 

125 be outside the image dimensions. 

126 Args: 

127 x (int): Initial x. 

128 x_step (int): x step that is the horizontal 

129 length of the subpart of the image. 

130 width (int): The width of the image. 

131 y (int): Initial y. 

132 y_step (int): y step that is the vertical length 

133 of the subpart of the image. 

134 height (int): The height of the image. 

135 

136 Returns: 

137 x1 (int): width value of the left edge of the selected part of the image. 

138 y1 (int): height value of the upper edge of the selected part of the image. 

139 x2 (int): width value of the right edge of the selected part of the image. 

140 y2 (int): height value of the lower edge of the selected part of the image. 

141 """ 

142 x1 = x 

143 x2 = x + x_step - 1 

144 if x2 >= width: 

145 x2 = width - 1 

146 y1 = y 

147 y2 = y + y_step - 1 

148 if y2 >= height: 

149 y2 = height - 1 

150 return x1, x2, y1, y2

185 """
